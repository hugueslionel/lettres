function checkAnswer(selectedLetter, buttonElement) { if (isAnswering) return; const question = questions[currentQuestionIndex]; const isCorrect = selectedLetter === question.correctAnswer; const feedback = document.getElementById('feedback'); if (isCorrect) { isAnswering = true; document.querySelectorAll('.choice-btn').forEach(btn => { btn.classList.add('disabled'); }); buttonElement.classList.add('correct'); feedback.textContent = `✅ Correct ! C'est bien la lettre "${question.letter.toUpperCase()}"`; feedback.className = 'feedback correct'; correctAnswers++; score += 10; playLetterSound(question.letter); // Passer automatiquement à la question suivante après le son + 1 seconde setTimeout(() => { nextQuestion(); }, 2000); // Durée approximative du son (1s) + 1 seconde de délai } else { buttonElement.classList.add('wrong'); feedback.textContent = `❌ Ce n'est pas la bonne lettre, réessaie !`; feedback.className = 'feedback wrong'; playWrongAnswerSound(selectedLetter); } document.getElementById('score').textContent = score; }
